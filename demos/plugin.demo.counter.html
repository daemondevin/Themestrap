4 · Custom formatter (jQuery init)
0
Bytes (human-readable)
$('#counter-custom').themestrapPluginCounter({
    to: 1073741824,
    speed: 3500,
    refreshInterval: 60,
    formatter(value) {
        const units = ['B','KB','MB','GB'];
        let u = 0;
        while (value >= 1024 && u < units.length - 1) { value /= 1024; u++; }
        return value.toFixed(2) + ' ' + units[u];
    }
});

5 · Programmatic API — replay / setValue / stop
0
Interactive counter
const inst = $('#counter-api').data('__pluginCounter');
inst.replay();          // re-run from current value to original target
inst.replay(500);       // count to a new target
inst.stop();            // freeze at current value
inst.setValue(999);     // jump instantly
