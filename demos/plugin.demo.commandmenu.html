Themestrap PluginCommandMenu Live Demo

A command palette
for the Themestrap way.

Press ⌘K (or Ctrl K) anywhere on this page to summon the menu. Type to filter, navigate with arrow keys, hit ↵ to execute. Built as a vanilla jQuery plugin — no React, no headless wrappers, no build step.

Also try ↑ ↓ to navigate, ↵ to select, esc to close.

Four different shapes, one plugin.

The same PluginCommandMenu drives all four examples below. Each just feeds it different items and different options. Click any trigger to see it in action.

Nav

Site navigation palette

Three groups — Navigation, Actions, Help — searchable across labels and keywords. Recently-selected items pin themselves to the top via localStorage. Same menu the hero ⌘K trigger summons.

Actions

Document actions menu

Action items dispatch real JS callbacks via registerAction() — no URL navigation. The toolbar below reflects state changes; selecting "Toggle starred" or "Archive" updates the document panel live.

Search

Slash-key launcher

Press / anywhere outside a text field to open a focused search palette. Same plugin, different shortcut grammar, different placeholder. GitHub-style.

Theme

Theme switcher palette

A tiny palette with four accent options. The active accent updates on selection via a JS callback that swaps CSS variables on the document root — watch the orange highlights change color everywhere.

$(function () { /* Initialise the menus that carry their options in data-plugin-options. The main-cmd has none, so we init it separately below with recents. */ $('[data-plugin-command-menu]:not(.manual)').not('#main-cmd').each(function () { const $this = $(this); const opts = themestrap.fn.getOptions($this.data('plugin-options')) || undefined; $this.themestrapPluginCommandMenu(opts); }); /* The main menu uses recents — pass that option explicitly */ $('#main-cmd').themestrapPluginCommandMenu({ recent: true, recentLimit: 4, }); const main = $('#main-cmd').data('__pluginCommandMenu'); /* Event log ─────────────────────────────────────────────*/ const $log = $('#event-log'); let logCount = 0; function fmtTime() { const d = new Date(); return d.toTimeString().slice(0, 8); } function logEvent(menuId, evtName, detail) { if (logCount === 0) $log.empty(); const $line = $('
').addClass('log-line'); $line.append($('').addClass('log-time').text(fmtTime())); $line.append($('').addClass('log-event-' + evtName).text('[' + menuId + ']')); $line.append(' '); $line.append($('').addClass('log-event-' + evtName).text('command:' + evtName)); if (detail) { $line.append(' '); $line.append($('').addClass('log-detail').text(detail)); } $log.prepend($line); logCount++; if (logCount > 50) { $log.children().slice(50).remove(); } } /* Subscribe to every menu — same handler, parametised by id */ $('[data-plugin-command-menu]').each(function () { const id = $(this).attr('id'); $(this) .on('command:open', () => logEvent(id, 'open')) .on('command:close', () => logEvent(id, 'close')) .on('command:filter', (e, inst, q, n) => { if (q !== '') logEvent(id, 'filter', '"' + q + '" → ' + n + ' match' + (n === 1 ? '' : 'es')); }) .on('command:select', (e, inst, $item, payload) => { logEvent(id, 'select', payload.label + (payload.action ? ' (action: ' + payload.action + ')' : '')); }); }); $('#clear-log').on('click', () => { $log.empty().append($('
').text('Cleared. Open a palette to start logging again.')); logCount = 0; }); /* Action callbacks ──────────────────────────────────────*/ /* Main menu — nav + action + help. The select event is already logged by the global handler above, so these callbacks can be no-ops in the demo. Registering them demonstrates that the plugin invokes them even when there's no href. */ [ 'demo-nav-home', 'demo-nav-projects', 'demo-nav-settings', 'demo-action-new', 'demo-action-invite', 'demo-action-upload', 'demo-help-docs', 'demo-help-support', 'demo-help-shortcuts', ].forEach(key => { main.registerAction(key, () => { /* logged by global handler */ }); }); /* Document actions menu — these have visible side effects */ const actions = $('#actions-cmd').data('__pluginCommandMenu'); let starred = false; let docStatus = 'draft · not starred'; function refreshDocStatus() { $('#doc-status').text(docStatus + (starred ? ' · ★ starred' : '')); } actions.registerAction('toggle-star', () => { starred = !starred; refreshDocStatus(); }); actions.registerAction('publish', () => { docStatus = 'published'; refreshDocStatus(); }); actions.registerAction('archive', () => { docStatus = 'archived'; refreshDocStatus(); }); actions.registerAction('rename', () => { const next = prompt('New document name:', $('#doc-title').text()); if (next) $('#doc-title').text(next); }); actions.registerAction('delete', () => { if (confirm('Delete the document? This is just a demo — nothing is actually deleted.')) { $('#doc-title').text('(deleted)'); docStatus = 'deleted'; starred = false; refreshDocStatus(); } }); /* Theme switcher */ const theme = $('#theme-cmd').data('__pluginCommandMenu'); const themeColors = { orange: '#e8672a', teal: '#2ab8c8', violet: '#a472ff', lime: '#5dd982', }; function applyTheme(name) { const c = themeColors[name]; if (!c) return; document.documentElement.style.setProperty('--brand-orange', c); document.documentElement.style.setProperty('--brand-orange-2', c); $('#theme-status').text(name); } theme.registerAction('theme-orange', () => applyTheme('orange')); theme.registerAction('theme-teal', () => applyTheme('teal')); theme.registerAction('theme-violet', () => applyTheme('violet')); theme.registerAction('theme-lime', () => applyTheme('lime')); /* Search menu — all actions just log via global handler */ const search = $('#search-cmd').data('__pluginCommandMenu'); ['search-docs', 'search-api', 'search-snippet', 'search-recipes'].forEach(key => { search.registerAction(key, () => { /* logged by global handler */ }); }); });
