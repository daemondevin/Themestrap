      Custom Validation Function

      validate: fn
        Choose a username
        Accept the terms
        Account ready

        Custom validation passed on both steps.

        $('#wiz-custom-instance').themestrapPluginWizard({
    validate: function ($step, stepNum) {
        if (stepNum === 1) {
            const val = $('#custom-username').val().trim();
            const ok  = /^[a-zA-Z0-9]{3,16}$/.test(val);
            $('#custom-username-error')
                .text(ok ? '' : 'Must be 3–16 alphanumeric characters.')
                .toggle(!ok);
            $('#custom-username').toggleClass('wiz-invalid', !ok);
            return ok;
        }
        if (stepNum === 2) {
            const ok = $('#custom-tos').prop('checked') && $('#custom-privacy').prop('checked');
            $('#custom-tos-error')
                .text(ok ? '' : 'Both checkboxes are required.')
                .toggle(!ok);
            return ok;
        }
        return true;
    }
});

          const wiz = $('#wiz-api-instance').data('__pluginWizard');

wiz.next();           // advance (runs validation)
wiz.prev();           // go back
wiz.goTo(3);          // jump to step 3
wiz.markComplete(2);  // manually mark step 2 done
wiz.getStep();        // → 3   (current step, 1-indexed)
wiz.getStepCount();   // → 5
wiz.isFirst();        // → false
wiz.isLast();         // → false
wiz.reset();          // return to step 1, clear completions

$('#wiz-api-instance').on('wizard:change', function (e, detail, from, to, dir) {
    console.log('moved', from, '→', to, '(', dir, ')');
});
$('#wiz-api-instance').on('wizard:complete', function () {
    console.log('wizard finished!');
});

        // call e.preventDefault() to block the transition

        console.log('changed to step', to, '(', dir, ')');
    })
    .on('wizard:invalid', function (e, inst, stepNum) {
        console.log('validation failed on step', stepNum);
    })
    .on('wizard:complete', function (e, inst) {
        console.log('wizard completed!');
    })
    .on('wizard:reset', function (e, inst) {
        console.log('wizard reset');
