Events will appear here…
$('#pt-events').on('ts-pricingtoggle-changed', (e, detail) => {
  console.log(detail.state);       // 'primary' | 'secondary'
  console.log(detail.isPrimary);   // boolean
  console.log(detail.isSecondary); // boolean
  console.log(detail.instance);    // PluginPricingToggle instance
});
